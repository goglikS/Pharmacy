create table MEDICATION
(
    MEDICATION_CODE NUMBER(5) ,
    M_NAME          VARCHAR2(100),
    FULL_NAME       VARCHAR2(200),
    MANUF_ID  		NUMBER(4),
    MANUF_DATE   DATE,
    CATEGORY_CODE NUMBER(2) ,
    PRESCRIPTION   NUMBER(1),
    PRICE         NUMBER(4,2),
    STOCK         NUMBER(5),
    

    CONSTRAINT MEDICATION_MEDICATION_CODE_pk PRIMARY KEY(MEDICATION_CODE),

    CONSTRAINT MEDICATION_MANUF_ID_fk FOREIGN KEY(MANUF_ID)
  REFERENCES MANUFACTURER (MANUF_ID),

    CONSTRAINT MEDICATION_CATEGORY_CODE_fk FOREIGN KEY(CATEGORY_CODE)
  REFERENCES CATEGORY (CATEGORY_CODE)                            
     
  );


CREATE TABLE MANUFACTURER

(
       MANUF_ID  NUMBER(4),
       MANUF_NAME VARCHAR2(50),
       COUNTRY_CODE NUMBER(6) ,
       CITY         VARCHAR2(28),
       ADRESS       varchar2(70),
       ZIP_CODE     NUMBER(5),
       TELEPHONE    NUMBER(15) ,

   

      CONSTRAINT MANUFACTURER_MANUF_ID_pk PRIMARY KEY (MANUF_ID),

  CONSTRAINT MANUFACTURER_COUNTRY_CODE_fk FOREIGN KEY(COUNTRY_CODE)
  REFERENCES COUNTRY (COUNTRY_CODE)    
);

CREATE TABLE SALES
(
       SALES_ID           NUMBER(10),
       MEDICATION_CODE    NUMBER(6) ,
       SALES_DATE         DATE,
       QUANTITY           NUMBER(3) ,
       DISCOUNT           NUMBER(2),
       FINAL_PRICE        NUMBER(5,2)  ,



      CONSTRAINT SALES_SALES_ID_pk PRIMARY KEY(SALES_ID),
  
   CONSTRAINT SALES_MEDICATION_CODE_fk FOREIGN KEY(MEDICATION_CODE)
  REFERENCES MEDICATION(MEDICATION_CODE) 

);







create table CATEGORY
(
    CATEGORY_CODE NUMBER(2) ,
    CATEGORY     VARCHAR2(50) ,
   CONSTRAINT CATEGORY_CATEGORY_CODE_pk PRIMARY KEY (CATEGORY_CODE)       
     
  );

create table COUNTRY
(
    COUNTRY_CODE NUMBER(2) ,
    COUNTRY     VARCHAR2(50) ,

   CONSTRAINT COUNTRY_COUNTRY_CODE_pk PRIMARY KEY(COUNTRY_CODE)             
     
  );
